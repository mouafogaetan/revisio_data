Files d'attente avec Bull + Redis

Envoyer des emails en background, générer des factures PDF et créer des rapports sans bloquer l'utilisateur

🐂 Bull 📧 Emails asynchrones 📄 PDF en arrière-plan 📊 Rapports
👆 Glissez ou utilisez les boutons 👆

🎯 Objectifs

  • ✔️ Comprendre l'utilité des files d'attente asynchrones
  • ✔️ Installer et configurer @nestjs/bull avec Redis
  • ✔️ Envoyer des emails de confirmation de commande en background
  • ✔️ Générer des factures PDF et des rapports via des workers
💡 Pourquoi ? Les opérations longues (envoi d'emails, génération PDF) ne doivent pas bloquer la réponse HTTP.
Concepts

📬 Qu'est-ce qu'une file d'attente ?

Principe : Producteur → Queue → Consommateur (Worker)

📤

Producteur

API NestJS qui ajoute un job

→
📋

Queue (Redis)

Stockage temporaire des jobs

→
⚙️

Worker

Processeur qui exécute le job

✅ Avantage 1 : Réponse HTTP immédiate
✅ Avantage 2 : Reprise sur échec (retry)
✅ Avantage 3 : Scalabilité horizontale
✅ Avantage 4 : Traitement par lots
Setup technique

📦 Installation de Bull

# Installation des dépendances
npm install @nestjs/bull bull
npm install @types/bull

# Redis doit être installé et en cours d'exécution
# docker run -d -p 6379:6379 redis

# Configuration dans app.module.ts
import { BullModule } from '@nestjs/bull';

@Module({
  imports: [
    BullModule.forRoot({
      redis: {
        host: 'localhost',
        port: 6379,
      },
    }),
    BullModule.registerQueue({
      name: 'email',
    }),
    BullModule.registerQueue({
      name: 'invoice',
    }),
  ],
})
export class AppModule {}
⚠️ Important : Redis doit être accessible depuis votre application NestJS.

📧 Queue d'envoi d'emails

// email.processor.ts
import { Processor, Process } from '@nestjs/bull';
import { Job } from 'bull';

@Processor('email')
export class EmailProcessor {
  @Process('send-confirmation')
  async handleOrderConfirmation(job: Job) {
    const { email, orderId, customerName, total } = job.data;
    
    console.log(`Envoi d'email à ${email} pour la commande ${orderId}`);
    
    // Simulation d'envoi d'email
    await this.emailService.send({
      to: email,
      subject: `Confirmation commande #${orderId}`,
      html: `<h1>Merci ${customerName} !</h1>
             <p>Votre commande de ${total} FCFA est confirmée.</p>`
    });
    
    return { success: true, orderId };
  }
}

// order.service.ts
async createOrder(cart: Cart) {
  const order = await this.orderRepository.save(newOrder);
  
  // Ajouter un job dans la queue (non bloquant)
  await this.emailQueue.add('send-confirmation', {
    email: order.user.email,
    orderId: order.id,
    customerName: order.user.firstName,
    total: order.total
  });
  
  return order; // Réponse immédiate sans attendre l'email
}

📄 Génération de PDF en arrière-plan

// invoice.processor.ts
@Processor('invoice')
export class InvoiceProcessor {
  @Process('generate')
  async generateInvoice(job: Job) {
    const { orderId, userId } = job.data;
    
    // 1. Récupérer les données de la commande
    const order = await this.orderService.findOne(orderId);
    
    // 2. Générer le PDF (opération longue)
    const pdfBuffer = await this.pdfService.generateOrderInvoice(order);
    
    // 3. Stocker sur S3 ou disque
    const fileUrl = await this.storageService.upload(
      `invoices/${orderId}.pdf`,
      pdfBuffer
    );
    
    // 4. Mettre à jour la commande avec l'URL
    await this.orderService.updateInvoiceUrl(orderId, fileUrl);
    
    // 5. Notifier l'utilisateur
    await this.emailQueue.add('send-notification', {
      userId,
      subject: 'Facture disponible',
      message: `Votre facture est disponible : ${fileUrl}`
    });
    
    return { fileUrl };
  }
}

// invoice.controller.ts
@Post('orders/:id/invoice')
@UseGuards(JwtAuthGuard)
async generateInvoice(@Param('id') orderId: string, @Req() req) {
  const job = await this.invoiceQueue.add('generate', {
    orderId,
    userId: req.user.id
  });
  
  return {
    message: 'Génération de la facture en cours',
    jobId: job.id,
    statusUrl: `/api/invoices/${job.id}/status`
  };
}
Robustesse

🔄 Gestion des échecs et reprises

@Processor('email')
export class EmailProcessor {
  @Process({
    name: 'send',
    concurrency: 5, // 5 jobs simultanés
  })
  async handleEmail(job: Job) {
    try {
      // Tentative d'envoi
      await this.emailService.send(job.data);
    } catch (error) {
      // Réessayer jusqu'à 3 fois avec délai exponentiel
      if (job.attemptsMade < 3) {
        throw error; // Bull retentera automatiquement
      }
      // Après 3 échecs, marquer comme failed
      await this.logFailedEmail(job.data, error);
    }
  }
}

// Ajout d'un job avec options
await this.emailQueue.add('send', data, {
  delay: 5000,           // Délai avant exécution (ms)
  attempts: 3,           // Nombre de tentatives
  backoff: {
    type: 'exponential', // Délai exponentiel entre les tentatives
    delay: 1000,         // Délai initial
  },
  removeOnComplete: true, // Supprimer le job après succès
  removeOnFail: false,    // Garder les jobs échoués pour analyse
});
