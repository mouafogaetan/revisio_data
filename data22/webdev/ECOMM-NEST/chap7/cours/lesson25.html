💡 Pourquoi ? Ajouter un nouveau moyen de paiement sans toucher au code existant (Open/Closed Principle).

🧩 Le problème : code spaghetti

Sans pattern, on finit avec des if/else ou switch monstres :

// ❌ Mauvaise approche
if (paymentMethod === 'stripe') {
  // 50 lignes spécifiques Stripe
} else if (paymentMethod === 'paypal') {
  // 50 lignes PayPal
} else if (paymentMethod === 'bank') {
  // 30 lignes virement
}

✅ Solution : Strategy Pattern

Une interface commune, chaque provider implémente sa propre stratégie. Le contrôleur choisit la stratégie au runtime.

UML simplifié : PaymentStrategy ← StripeStrategy, PayPalStrategy, BankTransferStrategy

📐 Interface abstraite

// payment-strategy.interface.ts
export interface PaymentStrategy {
  createPayment(amount: number, currency: string, metadata?: any): Promise<PaymentResult>;
  verifyPayment(paymentId: string): Promise<PaymentStatus>;
  refundPayment(paymentId: string, amount?: number): Promise<RefundResult>;
}

export interface PaymentResult {
  success: boolean;
  transactionId: string;
  redirectUrl?: string;
  clientSecret?: string;  // Pour Stripe
  errorMessage?: string;
}

export enum PaymentStatus {
  PENDING = 'pending',
  SUCCESS = 'success',
  FAILED = 'failed',
  REFUNDED = 'refunded'
}
📌 Tous les providers DOIVENT implémenter ces méthodes.

💳 Implémentation Stripe

// stripe.strategy.ts
import Stripe from 'stripe';
import { PaymentStrategy, PaymentResult } from './interfaces';

export class StripeStrategy implements PaymentStrategy {
  private stripe: Stripe;

  constructor() {
    this.stripe = new Stripe(process.env.STRIPE_SECRET_KEY);
  }

  async createPayment(amount: number, currency: string, metadata?: any): Promise<PaymentResult> {
    try {
      const paymentIntent = await this.stripe.paymentIntents.create({
        amount: Math.round(amount * 100), // Stripe utilise les cents
        currency,
        metadata,
        payment_method_types: ['card']
      });
      return {
        success: true,
        transactionId: paymentIntent.id,
        clientSecret: paymentIntent.client_secret,
        redirectUrl: null
      };
    } catch (error) {
      return { success: false, transactionId: null, errorMessage: error.message };
    }
  }

  async verifyPayment(paymentId: string): Promise<PaymentStatus> {
    const pi = await this.stripe.paymentIntents.retrieve(paymentId);
    if (pi.status === 'succeeded') return PaymentStatus.SUCCESS;
    if (pi.status === 'failed') return PaymentStatus.FAILED;
    return PaymentStatus.PENDING;
  }
}

    const reference = `VIR${Date.now()}`;

      transactionId: reference,
      redirectUrl: null,
      // Information supplémentaire pour le client
      metadata: { iban: 'CM211000...', reference }
    };
  }
}

⚙️ Le contexte : PaymentService

// payment.service.ts
@Injectable()
export class PaymentService {
  private strategies: Map<string, PaymentStrategy> = new Map();

  constructor(
    private stripeStrategy: StripeStrategy,
    private paypalStrategy: PayPalStrategy,
    private bankStrategy: BankTransferStrategy
  ) {
    this.strategies.set('stripe', stripeStrategy);
    this.strategies.set('paypal', paypalStrategy);
    this.strategies.set('bank', bankStrategy);
  }

  async processPayment(method: string, amount: number, currency: string, orderId: string) {
    const strategy = this.strategies.get(method);
    if (!strategy) throw new Error('Méthode de paiement inconnue');
    
    const result = await strategy.createPayment(amount, currency, { orderId });
    // Sauvegarde en base (voir slide suivant)
    return result;
  }
}

💾 Enregistrement des transactions

// payment.entity.ts
@Entity()
export class Payment {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column()
  orderId: string;

  @Column()
  method: string; // 'stripe', 'paypal', 'bank'

  @Column('decimal', { precision: 10, scale: 2 })
  amount: number;

  @Column()
  currency: string;

  @Column({ nullable: true })
  transactionId: string; // ID retourné par le provider

  @Column({ type: 'enum', enum: ['pending', 'success', 'failed', 'refunded'] })
  status: string;

  @Column({ type: 'json', nullable: true })
  rawResponse: any; // Réponse brute du provider (pour debug)

  @Column({ type: 'timestamp', default: () => 'CURRENT_TIMESTAMP' })
  createdAt: Date;
}
✅ Chaque paiement est enregistré, quelle que soit la méthode.

🌐 Contrôleur Order / Payment

// order.controller.ts
@Post(':id/pay')
@UseGuards(JwtAuthGuard)
async payOrder(@Param('id') orderId: string, @Body('method') method: string) {
  const order = await this.orderService.findOne(orderId);
  // Vérifier que la commande appartient bien à l'utilisateur
  
  const paymentResult = await this.paymentService.processPayment(
    method,
    order.total,
    order.currency,
    orderId
  );
  
  // Sauvegarder le paiement
  await this.paymentRepository.save({
    orderId,
    method,
    amount: order.total,
    transactionId: paymentResult.transactionId,
    status: paymentResult.success ? 'pending' : 'failed'
  });
  
  if (paymentResult.redirectUrl) {
    return { redirect_url: paymentResult.redirectUrl };
  }
  return { client_secret: paymentResult.clientSecret };
}

// payment.service.ts
this.strategies.set('mobile_money', new MobileMoneyStrategy());
