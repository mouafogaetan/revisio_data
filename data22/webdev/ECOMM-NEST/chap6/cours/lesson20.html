Leçon 20 : Panier → Commande

Du panier à la commande

Transformation, stock, email de confirmation

📦 Chapitre 6•⚡ Leçon 20•✉️ Confirmation

Objectifs de la leçon

  • ✅ Transformer le panier en commande (Order + OrderItem)
  • ✅ Décrémenter les stocks réels après validation
  • ✅ Envoyer un email de confirmation au client (Nodemailer)
  • ✅ Gérer les transactions pour garantir l'intégrité
💡 La validation de commande est une opération critique : transaction ACID, vérification de stock, email asynchrone.

📋 1. Entités Order et OrderItem

// order.entity.ts
@Entity()
export class Order {
    @PrimaryGeneratedColumn('uuid')
    id: string;

    @ManyToOne(() => User)
    user: User;

    @Column({ type: 'enum', enum: OrderStatus, default: OrderStatus.PENDING })
    status: OrderStatus;

    @Column('decimal', { precision: 10, scale: 2 })
    total: number;

    @Column()
    shippingAddress: string;

    @Column({ nullable: true })
    trackingNumber: string;

    @OneToMany(() => OrderItem, item => item.order, { cascade: true })
    items: OrderItem[];

    @CreateDateColumn()
    createdAt: Date;
}

📸 2. OrderItem : snapshot produit

// order-item.entity.ts
@Entity()
export class OrderItem {
    @PrimaryGeneratedColumn('uuid')
    id: string;

    @ManyToOne(() => Order, order => order.items)
    order: Order;

    @Column()
    productName: string;      // Snapshot (nom au moment de l'achat)

    @Column('decimal')
    priceAtPurchase: number;  // Prix au moment de l'achat

    @Column()
    quantity: number;

    @Column({ type: 'json', nullable: true })
    variation: Record;

    @Column({ nullable: true })
    productImage: string;
}
⚠️ On ne lie pas directement Product pour éviter les modifications futures (prix, nom). On stocke une copie (snapshot).

🔄 3. Transformation panier → commande

async checkout(userId: string, checkoutDto: CheckoutDto): Promise<Order> {
    const cart = await this.cartService.getCartByUserId(userId);
    if (!cart.items.length) throw new BadRequestException('Panier vide');
    
    // Vérifier les stocks
    for (const item of cart.items) {
        if (item.product.stock < item.quantity) {
            throw new BadRequestException(`Stock insuffisant pour ${item.product.name}`);
        }
    }
    
    // Créer la commande
    const order = this.orderRepository.create({
        user: { id: userId },
        shippingAddress: checkoutDto.shippingAddress,
        total: await this.cartService.getCartTotal(cart.id),
        items: cart.items.map(item => this.orderItemRepository.create({
            productName: item.product.name,
            priceAtPurchase: item.product.price,
            quantity: item.quantity,
            variation: item.variation,
            productImage: item.product.images?.[0]
        }))
    });
    
    return this.orderRepository.save(order);
}

📉 4. Décrémentation des stocks (transaction)

async checkout(userId: string, checkoutDto: CheckoutDto): Promise<Order> {
    // Utiliser une transaction TypeORM
    return this.dataSource.transaction(async (manager) => {
        const cart = await this.cartService.getCartWithItems(cartId, manager);
        
        // Décrémenter les stocks
        for (const item of cart.items) {
            const product = await manager.findOne(Product, { where: { id: item.product.id }, lock: { mode: 'pessimistic_write' } });
            if (product.stock < item.quantity) throw new Error(`Stock insuffisant`);
            product.stock -= item.quantity;
            await manager.save(product);
        }
        
        // Créer la commande
        const order = manager.create(Order, { ... });
        await manager.save(order);
        
        // Vider le panier
        await manager.delete(CartItem, { cart: { id: cart.id } });
        
        return order;
    });
}

✉️ 5. Email de confirmation

// OrderService
async sendOrderConfirmation(order: Order) {
    const user = await this.userService.findOne(order.user.id);
    const html = `
        <h1>Merci pour votre commande #${order.id}</h1>
        <p>Bonjour ${user.firstName},</p>
        <p>Votre commande d'un montant de ${order.total} FCFA a été confirmée.</p>
        <h2>Détails :</h2>
        <ul>${order.items.map(i => `<li>${i.productName} x${i.quantity} - ${i.priceAtPurchase} FCFA</li>`).join('')}</ul>
        <p>Nous vous tiendrons informé de l'expédition.</p>
    `;
    await this.emailService.send({
        to: user.email,
        subject: `Confirmation de commande #${order.id}`,
        html
    });
}

⚙️ 6. Configuration Nodemailer

// email.service.ts
import * as nodemailer from 'nodemailer';

@Injectable()
export class EmailService {
    private transporter;
    
    constructor(private configService: ConfigService) {
        this.transporter = nodemailer.createTransport({
            host: this.configService.get('SMTP_HOST'),
            port: this.configService.get('SMTP_PORT'),
            secure: false,
            auth: {
                user: this.configService.get('SMTP_USER'),
                pass: this.configService.get('SMTP_PASS'),
            },
        });
    }
    
    async send(options: { to: string; subject: string; html: string }) {
        await this.transporter.sendMail({
            from: `"E-commerce" <${this.configService.get('SMTP_FROM')}>`,
            ...options
        });
    }
}

🌐 7. Endpoint de validation

// OrderController
@Post('checkout')
@UseGuards(JwtAuthGuard)
async checkout(@Req() req, @Body() checkoutDto: CheckoutDto) {
    const order = await this.orderService.checkout(req.user.id, checkoutDto);
    
    // Envoi asynchrone (ne pas bloquer la réponse)
    this.orderService.sendOrderConfirmation(order).catch(err => 
        this.logger.error(`Erreur envoi email: ${err.message}`)
    );
    
    return {
        message: 'Commande validée',
        orderId: order.id,
        total: order.total
    };
}
📌 L'envoi d'email est asynchrone pour ne pas ralentir la réponse utilisateur (utiliser Bull/Queue en production).
