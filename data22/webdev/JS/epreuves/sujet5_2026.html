    5 Cas pratique : Gestion d'une liste de tâches (To-Do List)

    13. Implémentez en JavaScript une To-Do List simple avec les fonctionnalités :

    • Un champ texte et un bouton "Ajouter" pour ajouter une tâche.
    • Un tableau (array) pour stocker les tâches en mémoire.
    • Affichage dynamique des tâches dans une liste HTML (ul).
    • Chaque tâche doit avoir un bouton "Supprimer" pour la retirer.
    • Utilisez map() pour générer le HTML des tâches.
    // Structure attendue (à compléter)
let tasks = [];
const taskInput = document.getElementById('taskInput');
const addBtn = document.getElementById('addBtn');
const taskList = document.getElementById('taskList');

function renderTasks() {
  taskList.innerHTML = tasks.map((task, index) => 
    `
  • ${task} 
    • `
  ).join('');
}

addBtn.addEventListener('click', () => {
  if (taskInput.value.trim()) {
    tasks.push(taskInput.value);
    renderTasks();
    taskInput.value = '';
  }
});

taskList.addEventListener('click', (e) => {
  if (e.target.tagName === 'BUTTON') {
    const idx = e.target.getAttribute('data-index');
    tasks.splice(idx, 1);
    renderTasks();
  }
});
