🛒 Implémentation de la page produit et du panier

Voici un extrait de code pour une page produit avec ajout au panier. Analysez-le et répondez aux questions.

import { useState } from 'react';
import { View, Text, Button, Alert } from 'react-native';
import { useCartStore } from '../store/cartStore';

export default function ProductDetailScreen({ route }) {
  const { product } = route.params;
  const [quantity, setQuantity] = useState(1);
  const addToCart = useCartStore((state) => state.addItem);
  const isInCart = useCartStore((state) => 
    state.items.some(item => item.id === product.id)
  );

  const handleAddToCart = () => {
    addToCart({ ...product, quantity });
    Alert.alert('Succès', `${product.name} ajouté au panier`);
  };

  return (
    <View>
      <Text>{product.name}</Text>
      <Text>Prix : {product.price} FCFA</Text>
      {!isInCart ? (
        <Button title={`Ajouter (${quantity})`} onPress={handleAddToCart} />
      ) : (
        <Button title="Déjà dans le panier" disabled />
      )}
    </View>
  );
}
