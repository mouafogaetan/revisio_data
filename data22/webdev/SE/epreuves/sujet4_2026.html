4 Maintenance & Dette technique (5 pts)

Analysez le code ci-dessous, identifiez les code smells et proposez un refactoring en appliquant les principes DRY et SRP.

class OrderProcessor {
    processOrder(order) {
        // Calcul des taxes
        let tax = 0;
        if (order.country === 'FR') tax = order.total * 0.2;
        else if (order.country === 'BE') tax = order.total * 0.21;
        else if (order.country === 'DE') tax = order.total * 0.19;
        else tax = order.total * 0.0;
        
        // Envoi d'email
        console.log(`Email envoyé à ${order.email}: Votre commande de ${order.total + tax}€ est confirmée`);
        
        // Sauvegarde en base
        console.log(`Commande ${order.id} sauvegardée`);
        
        // Mise à jour du stock
        order.items.forEach(item => {
            console.log(`Stock réduit pour ${item.productId}: -${item.quantity}`);
        });
        
        return { success: true, total: order.total + tax };
    }
}
