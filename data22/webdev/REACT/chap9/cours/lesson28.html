❌ Les limites de fetch + useEffect

Avant React Query, on utilisait souvent ce pattern :

function UserList() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch('/api/users')
      .then(res => res.json())
      .then(data => {
        setUsers(data);
        setLoading(false);
      })
      .catch(err => {
        setError(err);
        setLoading(false);
      });
  }, []);

  if (loading) return <div>Chargement...</div>;
  if (error) return <div>Erreur : {error.message}</div>;
  return <div>{users.map(...)}</div>;
}

Problèmes :

  • Pas de caching (même donnée rechargée à chaque montage)
  • Pas de revalidation automatique
  • Gestion manuelle du loading/error
  • Pas de retry automatique en cas d'échec
  • Données non partagées entre composants

📊 useQuery : La base

import { useQuery } from '@tanstack/react-query';

function UserList() {
  const { data, isLoading, error } = useQuery({
    queryKey: ['users'],
    queryFn: () => fetch('/api/users').then(res => res.json())
  });

  if (isLoading) return <div>Chargement...</div>;
  if (error) return <div>Erreur : {error.message}</div>;
  
  return (
    <ul>
      {data.map(user => <li key={user.id}>{user.name}</li>)}
    </ul>
  );
}

React Query gère tout : loading, erreur, cache, refetch !

⚙️ Options avancées de useQuery

staleTime: 5000

Les données sont considérées "fraîches" pendant 5 secondes (pas de refetch)

refetchInterval: 10000

Rafraîchissement automatique toutes les 10 secondes

enabled: isReady

Exécuter la requête seulement quand une condition est remplie

retry: 3

Réessayer 3 fois en cas d'échec

const { data } = useQuery({
  queryKey: ['user', userId],
  queryFn: () => fetchUser(userId),
  staleTime: 5 * 60 * 1000,  // 5 minutes
  refetchOnWindowFocus: true,  // Refetch quand l'onglet redevient actif
  retry: 2
});

✏️ useMutation : Créer, modifier, supprimer

import { useMutation, useQueryClient } from '@tanstack/react-query';

function AddUser() {
  const queryClient = useQueryClient();
  
  const mutation = useMutation({
    mutationFn: (newUser) => 
      fetch('/api/users', {
        method: 'POST',
        body: JSON.stringify(newUser),
        headers: { 'Content-Type': 'application/json' }
      }).then(res => res.json()),
    
    onSuccess: () => {
      // Invalider le cache pour recharger la liste
      queryClient.invalidateQueries({ queryKey: ['users'] });
    }
  });

  return (
    <button onClick={() => mutation.mutate({ name: 'Nouveau user' })}>
      {mutation.isPending ? 'Ajout en cours...' : 'Ajouter'}
    </button>
  );
}

Invalidation du cache : après une mutation, on indique à React Query quelles requêtes doivent être rechargées.

📊 États d'une mutation

isIdle

Mutation non encore exécutée

isPending

Mutation en cours (loading)

isSuccess

Mutation réussie

isError

Erreur lors de la mutation

const mutation = useMutation({ mutationFn: addTodo });

return (
  <div>
    {mutation.isPending && <p>Ajout en cours...</p>}
    {mutation.isError && <p>Erreur : {mutation.error.message}</p>}
    {mutation.isSuccess && <p>Todo ajouté !</p>}
    <button onClick={() => mutation.mutate(newTodo)}>
      Ajouter
    </button>
  </div>
);

📝 Exemple complet : Todo App avec React Query

// hooks/useTodos.js
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';

const fetchTodos = () => fetch('/api/todos').then(res => res.json());
const addTodo = (newTodo) => fetch('/api/todos', {
  method: 'POST',
  body: JSON.stringify(newTodo),
  headers: { 'Content-Type': 'application/json' }
}).then(res => res.json());

export function useTodos() {
  return useQuery({
    queryKey: ['todos'],
    queryFn: fetchTodos
  });
}

export function useAddTodo() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: addTodo,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] });
    }
  });
}

// Composant TodoList
function TodoList() {
  const { data: todos, isLoading, error } = useTodos();
  const addTodoMutation = useAddTodo();
  const [newTodo, setNewTodo] = useState('');

  if (isLoading) return <div>Chargement...</div>;
  if (error) return <div>Erreur : {error.message}</div>;

  return (
    <div>
      <input value={newTodo} onChange={(e) => setNewTodo(e.target.value)} />
      <button onClick={() => addTodoMutation.mutate({ title: newTodo })}>
        {addTodoMutation.isPending ? 'Ajout...' : 'Ajouter'}
      </button>
      <ul>
        {todos.map(todo => <li key={todo.id}>{todo.title}</li>)}
      </ul>
    </div>
  );
}

✨ Bonnes pratiques React Query

  • Organisation des queryKey : utiliser des clés structurées ['todos', 'user', userId]
  • Extraire les hooks : créer des hooks personnalisés useTodos(), useAddTodo()
  • Utiliser les DevTools : indispensables pour comprendre le cache
  • Configurer le QueryClient : defaultOptions pour des comportements globaux
  • Invalidation sélective : invalider uniquement les requêtes concernées
  • setQueryData pour les mises à jour optimistes : améliorer la réactivité

Mise à jour optimiste : mettre à jour le cache immédiatement, puis annuler si la mutation échoue.

🧪 Exercice : Galerie de photos

Créez une application qui affiche des photos depuis une API et permet d'en ajouter de nouvelles.

Consignes :

  1. Utiliser useQuery pour récupérer les photos (GET /api/photos)
  2. Afficher un loader et un message d'erreur
  3. Utiliser useMutation pour ajouter une nouvelle photo (POST /api/photos)
  4. Invalider la requête ['photos'] après l'ajout pour voir la nouvelle photo apparaître
  5. Bonus : ajouter un délai de rafraîchissement automatique toutes les 30 secondes
// Solution simplifiée
const { data, isLoading } = useQuery({
  queryKey: ['photos'],
  queryFn: () => fetch('/api/photos').then(res => res.json()),
  refetchInterval: 30000  // Bonus
});

const mutation = useMutation({
  mutationFn: (newPhoto) => fetch('/api/photos', {
    method: 'POST',
    body: JSON.stringify(newPhoto),
    headers: { 'Content-Type': 'application/json' }
  }),
  onSuccess: () => queryClient.invalidateQueries({ queryKey: ['photos'] })
});
