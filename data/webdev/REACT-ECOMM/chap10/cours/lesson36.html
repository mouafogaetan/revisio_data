📝 Structure du formulaire produit

Le formulaire doit gérer : nom, description, prix, stock, catégorie, images, et variations.

// ProductForm.jsx - Structure de base
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

const productSchema = z.object({
  name: z.string().min(3, 'Nom trop court'),
  description: z.string().min(10),
  price: z.number().positive(),
  stock: z.number().int().nonnegative(),
  categoryId: z.string().uuid(),
  images: z.array(z.string().url()).min(1, 'Au moins une image'),
  variations: z.array(z.object({
    size: z.string().optional(),
    color: z.string().optional(),
    sku: z.string(),
    stock: z.number().int().nonnegative()
  })).optional()
});

const ProductForm = ({ product, onSubmit }) => {
  const { register, handleSubmit, formState: { errors } } = useForm({
    resolver: zodResolver(productSchema),
    defaultValues: product || {}
  });

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
      {/* champs du formulaire */}
    </form>
  );
};

🖼️ Upload d'images avec preview

Composant ImageUploader avec drag & drop, preview et upload vers Cloudinary.

// components/ImageUploader.jsx
import { useState, useCallback } from 'react';
import { useDropzone } from 'react-dropzone';

const ImageUploader = ({ onUpload, existingImages = [] }) => {
  const [images, setImages] = useState(existingImages);
  const [uploading, setUploading] = useState(false);

  const onDrop = useCallback(async (acceptedFiles) => {
    setUploading(true);
    const formData = new FormData();
    acceptedFiles.forEach(file => formData.append('images', file));
    
    const response = await fetch('/api/admin/upload', {
      method: 'POST',
      body: formData
    });
    const { urls } = await response.json();
    const newImages = [...images, ...urls];
    setImages(newImages);
    onUpload(newImages);
    setUploading(false);
  }, [images, onUpload]);

  const { getRootProps, getInputProps, isDragActive } = useDropzone({
    onDrop,
    accept: { 'image/*': [] },
    maxFiles: 5
  });

  return (
    <div>
      <div {...getRootProps()} className={`border-2 border-dashed rounded-lg p-6 text-center cursor-pointer transition ${isDragActive ? 'border-indigo-500 bg-indigo-50' : 'border-gray-300'}`}>
        <input {...getInputProps()} />
        {isDragActive ? <p>Déposez les images ici...</p> : <p>Glissez-déposez des images, ou cliquez pour sélectionner</p>}
      </div>
      {uploading && <p>Upload en cours...</p>}
      <div className="flex gap-2 mt-3 flex-wrap">
        {images.map((url, idx) => (
          <div key={idx} className="relative w-20 h-20">
            <img src={url} className="w-full h-full object-cover rounded" />
            <button onClick={() => removeImage(idx)} className="absolute -top-2 -right-2 bg-red-500 text-white rounded-full w-5 h-5 text-xs">×</button>
          </div>
        ))}
      </div>
    </div>
  );
};

📊 Gestion des variations (taille, couleur)

Permet de gérer différentes déclinaisons d'un même produit avec stocks individuels.

// components/VariationsManager.jsx
import { useFieldArray } from 'react-hook-form';

const VariationsManager = ({ control }) => {
  const { fields, append, remove } = useFieldArray({
    control,
    name: 'variations'
  });

  return (
    <div className="space-y-3">
      <div className="flex justify-between items-center">
        <h3 className="font-semibold">Variations</h3>
        <button type="button" onClick={() => append({ size: '', color: '', sku: '', stock: 0 })} 
          className="text-sm bg-indigo-100 px-3 py-1 rounded">
          + Ajouter une variation
        </button>
      </div>
      
      {fields.map((field, index) => (
        <div key={field.id} className="grid grid-cols-4 gap-2 items-center border p-3 rounded">
          <input {...register(`variations.${index}.size`)} placeholder="Taille" className="border rounded p-1" />
          <input {...register(`variations.${index}.color`)} placeholder="Couleur" className="border rounded p-1" />
          <input {...register(`variations.${index}.sku`)} placeholder="SKU" className="border rounded p-1" />
          <input type="number" {...register(`variations.${index}.stock`)} placeholder="Stock" className="border rounded p-1" />
          <button type="button" onClick={() => remove(index)} className="text-red-500">🗑️</button>
        </div>
      ))}
    </div>
  );
};

📋 Liste des produits (admin)

// pages/admin/ProductsList.jsx
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { toast } from 'sonner';

const ProductsList = () => {
  const queryClient = useQueryClient();
  const { data: products, isLoading } = useQuery({
    queryKey: ['admin-products'],
    queryFn: () => api.get('/admin/products')
  });

  const deleteMutation = useMutation({
    mutationFn: (id) => api.delete(`/admin/products/${id}`),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin-products'] });
      toast.success('Produit supprimé');
    }
  });

  if (isLoading) return <div>Chargement...</div>

  return (
    <div className="overflow-x-auto">
      <table className="min-w-full bg-white rounded-lg">
        <thead>
          <tr>
            <th>Image</th><th>Nom</th><th>Prix</th><th>Stock</th><th>Actions</th>
          </tr>
        </thead>
        <tbody>
          {products?.map(product => (
            <tr key={product.id}>
              <td><img src={product.images[0]} className="w-10 h-10 object-cover" /></td>
              <td>{product.name}</td>
              <td>{product.price}€</td>
              <td>{product.stock}</td>
              <td>
                <Link to={`/admin/products/edit/${product.id}`} className="text-blue-500 mr-2">✏️</Link>
                <button onClick={() => deleteMutation.mutate(product.id)} className="text-red-500">🗑️</button>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
};

📦 Gestion des stocks et alertes

Afficher les produits en stock faible et permettre de mettre à jour rapidement les quantités.

// components/LowStockAlert.jsx
import { useQuery } from '@tanstack/react-query';

const LowStockAlert = () => {
  const { data: lowStockProducts } = useQuery({
    queryKey: ['low-stock'],
    queryFn: () => api.get('/admin/products/low-stock?threshold=5'),
    refetchInterval: 30000 // toutes les 30 secondes
  });

  if (!lowStockProducts?.length) return null;

  return (
    <div className="bg-yellow-50 border-l-4 border-yellow-500 p-4 mb-4 rounded">
      <h3 className="font-bold text-yellow-800">⚠️ Stock faible ({lowStockProducts.length} produits)</h3>
      <ul className="mt-2 text-sm">
        {lowStockProducts.map(p => (
          <li key={p.id}>{p.name} : {p.stock} unité(s) restante(s)</li>
        ))}
      </ul>
    </div>
  );
};

⚡ Mise à jour rapide du stock

// components/QuickStockUpdate.jsx
const QuickStockUpdate = ({ productId, currentStock }) => {
  const [stock, setStock] = useState(currentStock);
  const [updating, setUpdating] = useState(false);
  const queryClient = useQueryClient();

  const updateStock = async () => {
    setUpdating(true);
    try {
      await api.patch(`/admin/products/${productId}/stock`, { stock });
      queryClient.invalidateQueries({ queryKey: ['admin-products'] });
      queryClient.invalidateQueries({ queryKey: ['low-stock'] });
      toast.success('Stock mis à jour');
    } catch (error) {
      toast.error('Erreur');
    } finally {
      setUpdating(false);
    }
  };

  return (
    <div className="flex items-center gap-2">
      <input
        type="number"
        value={stock}
        onChange={(e) => setStock(parseInt(e.target.value))}
        className="w-20 border rounded p-1 text-center"
      />
      <button onClick={updateStock} disabled={updating} className="bg-green-500 text-white px-2 py-1 rounded text-sm">
        {updating ? '...' : 'Mettre à jour'}
      </button>
    </div>
  );
};

✅ Bonnes pratiques pour l'interface admin

  • Utiliser React Hook Form + Zod pour la validation des formulaires
  • Prévisualisation des images avant upload
  • Gérer l'état de chargement (loading) pour les mutations
  • Confirmation avant suppression (modal ou window.confirm)
  • Utiliser React Query pour le cache et la revalidation automatique
  • Séparer la logique métier dans des hooks personnalisés
  • Protéger les routes admin avec un guard côté frontend ET backend
