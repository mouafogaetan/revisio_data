⚖️ Calcul des frais selon le poids total

Chaque produit a un poids (kg). On additionne, puis on applique un barème.

// Dans le store cartStore.js
getTotalWeight: () => {
  const items = get().items;
  return items.reduce((sum, item) => sum + (item.weight * item.quantity), 0);
},

getShippingCost: () => {
  const weight = get().getTotalWeight();
  const subtotal = get().getSubtotal();
  
  // Livraison offerte dès 100€
  if (subtotal >= 100) return 0;
  
  // Barème basé sur le poids (exemple)
  if (weight <= 1) return 3.99;
  if (weight <= 3) return 5.99;
  if (weight <= 10) return 7.99;
  return 12.99;
}
Formule : \( \text{Poids total} = \sum (\text{poids}_i \times \text{quantité}_i) \)

📅 Options de livraison et estimation de date

// shippingOptions.js
export const shippingOptions = [
  { id: 'standard', name: 'Standard', days: 5, costMultiplier: 1 },
  { id: 'express', name: 'Express', days: 2, costMultiplier: 1.5 },
  { id: 'pickup', name: 'Retrait en magasin', days: 1, costMultiplier: 0 }
];

// Fonction d'estimation de date
function estimateDeliveryDate(shippingDays) {
  const date = new Date();
  date.setDate(date.getDate() + shippingDays);
  return date.toLocaleDateString('fr-FR', { day: 'numeric', month: 'long', year: 'numeric' });
}

💡 Exemple : Aujourd'hui + 5 jours ouvrables = 12 décembre 2025 (à adapter selon le mode).

🛍️ Composant de sélection de livraison

// components/ShippingSelector.jsx
import { useState } from 'react';
import { shippingOptions, estimateDeliveryDate } from '../utils/shipping';
import { useCartStore } from '../stores/cartStore';

const ShippingSelector = () => {
  const [selectedMethod, setSelectedMethod] = useState('standard');
  const baseCost = useCartStore(state => state.getShippingCost());
  const selected = shippingOptions.find(opt => opt.id === selectedMethod);
  const finalCost = baseCost * (selected?.costMultiplier || 1);
  const deliveryDate = estimateDeliveryDate(selected?.days || 5);

  return (
    <div className="border rounded-xl p-4">
      <h3 className="font-bold mb-3">Mode de livraison</h3>
      {shippingOptions.map(opt => (
        <label key={opt.id} className="flex items-center gap-3 mb-2 cursor-pointer">
          <input type="radio" name="shipping" value={opt.id} checked={selectedMethod === opt.id}
                 onChange={() => setSelectedMethod(opt.id)} />
          <span>{opt.name} – {opt.days} jours ouvrables</span>
          <span className="ml-auto font-semibold">
            {opt.costMultiplier === 0 ? 'Gratuit' : (baseCost * opt.costMultiplier).toFixed(2) + ' €'}
          </span>
        </label>
      ))}
      <p className="text-sm text-gray-500 mt-3">Livraison estimée : {deliveryDate}</p>
    </div>
  );
};
