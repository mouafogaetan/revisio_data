📦 Store Zustand pour le panier

Avant la page panier, voici le store qui gère les articles, quantités et actions.

// stores/cartStore.js
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

export const useCartStore = create(
  persist(
    (set, get) => ({
      items: [], // { id, name, price, image, quantity }
      updateQuantity: (id, quantity) => {
        if (quantity <= 0) {
          set({ items: get().items.filter(i => i.id !== id) });
        } else {
          set({
            items: get().items.map(i => i.id === id ? { ...i, quantity } : i)
          });
        }
      },
      removeItem: (id) => set({ items: get().items.filter(i => i.id !== id) }),
      clearCart: () => set({ items: [] }),
    }),
    { name: 'cart-storage' }
  )
);

🛒 Composant CartPage (partie 1) : tableau des produits

// pages/CartPage.jsx
import { useCartStore } from '../stores/cartStore';
import { Link } from 'react-router-dom';

const CartPage = () => {
  const { items, updateQuantity, removeItem } = useCartStore();
  
  if (items.length === 0) {
    return (
      <div className="text-center py-16">
        <h2 className="text-2xl font-semibold">Votre panier est vide</h2>
        <Link to="/products" className="text-emerald-600 underline">Continuer mes achats</Link>
      </div>
    );
  }

  return (
    <div className="max-w-6xl mx-auto">
      <h1 className="text-3xl font-bold mb-6">Mon panier ({items.length} articles)</h1>
      <div className="grid md:grid-cols-3 gap-8">
        {/* Liste des produits - 2/3 colonne */}
        <div className="md:col-span-2 space-y-4">
          {items.map(item => (
            <div key={item.id} className="flex gap-4 border-b pb-4">
              <img src={item.image} alt={item.name} className="w-24 h-24 object-cover rounded" />
              <div className="flex-1">
                <h3 className="font-semibold">{item.name}</h3>
                <p className="text-emerald-600 font-bold">{item.price} €</p>
                <div className="flex items-center gap-3 mt-2">
                  <select value={item.quantity} onChange={e => updateQuantity(item.id, +e.target.value)} className="border rounded px-2 py-1">
                    {[...Array(10).keys()].map(n => <option key={n+1} value={n+1}>{n+1}</option>)}
                  </select>
                  <button onClick={() => removeItem(item.id)} className="text-red-500 text-sm">Supprimer</button>
                </div>
              </div>
              <div className="font-bold">{(item.price * item.quantity).toFixed(2)} €</div>
            </div>
          ))}
        </div>
        {/* Récapitulatif des totaux - 1/3 colonne */}
      </div>
    </div>
  );
};

🧮 Calcul des montants : sous-total, TVA, frais de port

On ajoute une fonction de calcul dans le store (ou directement dans le composant).

// Dans le store (getters)
getTotalPrice: () => {
  const items = get().items;
  return items.reduce((sum, i) => sum + (i.price * i.quantity), 0);
},
getSubtotal: () => get().getTotalPrice(),
getTaxes: (rate = 0.1925) => get().getSubtotal() * rate,
getShipping: () => {
  const subtotal = get().getSubtotal();
  if (subtotal === 0) return 0;
  return subtotal > 100 ? 0 : 5.99; // Livraison offerte dès 100€
},
getTotal: () => get().getSubtotal() + get().getTaxes() + get().getShipping()

// Dans le composant CartPage
const subtotal = useCartStore(state => state.getSubtotal());
const taxes = useCartStore(state => state.getTaxes());
const shipping = useCartStore(state => state.getShipping());
const total = useCartStore(state => state.getTotal());
Formules : \( \text{Total} = \text{Sous-total} + \text{TVA} + \text{Livraison} \) avec \( \text{TVA} = \text{Sous-total} \times 0.1925 \) (Cameroun).
