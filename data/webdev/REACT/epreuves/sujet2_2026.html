/6 points

On souhaite afficher une galerie de profils. Complétez le code React et répondez aux questions.

// Composant Carte (à compléter)
function Carte({ nom, role, children }) {
  return (
    <div className="border rounded-lg p-4 shadow">
      <h3 className="font-bold">{/* afficher nom */}</h3>
      <p>{/* afficher role */}</p>
      <div className="mt-2 text-gray-600">
        {children}
      </div>
    </div>
  );
}

// Utilisation dans App
function App() {
  return (
    <div>
      <Carte nom="Alice Kamga" role="Développeuse Frontend">
        <span>⭐ 4.8 (12 avis)</span>
      </Carte>
    </div>
  );
}
