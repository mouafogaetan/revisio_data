🏠 1. Entité Address

Une adresse de livraison appartient à un seul utilisateur. Relation OneToMany.

// address.entity.ts
import { Entity, Column, ManyToOne, PrimaryGeneratedColumn } from 'typeorm';
import { User } from './user.entity';

@Entity()
export class Address {
    @PrimaryGeneratedColumn('uuid')
    id: string;

    @Column()
    street: string;

    @Column()
    city: string;

    @Column()
    postalCode: string;

    @Column()
    country: string;

    @Column({ default: false })
    isDefault: boolean;

    @ManyToOne(() => User, user => user.addresses)
    user: User;
}

👑 3. CRUD administrateur

// UserService - méthodes admin
async findAll(): Promise {
    return this.userRepository.find({ relations: ['addresses'] });
}

async findOne(id: string): Promise {
    const user = await this.userRepository.findOne({ 
        where: { id }, 
        relations: ['addresses'] 
    });
    if (!user) throw new NotFoundException();
    return user;
}

async update(id: string, dto: UpdateUserDto): Promise {
    await this.userRepository.update(id, dto);
    return this.findOne(id);
}

async remove(id: string): Promise {
    await this.userRepository.delete(id);
}

✏️ 4. Mise à jour du profil

L'utilisateur authentifié peut modifier ses propres informations (sauf email et rôles).

// UserController
@Put('profile')
@UseGuards(JwtAuthGuard)
async updateProfile(@Req() req, @Body() dto: UpdateProfileDto) {
    const userId = req.user.id;
    return this.userService.updateProfile(userId, dto);
}

// UserService
async updateProfile(userId: string, dto: UpdateProfileDto) {
    // Empêcher la modification de l'email et des rôles
    const { email, roles, ...safeData } = dto;
    await this.userRepository.update(userId, safeData);
    return this.findOne(userId);
}

// AddressService
async create(userId: string, dto: CreateAddressDto) {
    const user = await this.userService.findOne(userId);
    const address = this.addressRepository.create({ ...dto, user });
    if (dto.isDefault) {
        // Désactiver les autres adresses par défaut
        await this.addressRepository.update(
            { user: { id: userId }, isDefault: true },
            { isDefault: false }
        );
    }
    return this.addressRepository.save(address);
}

@Put('profile')
@UseGuards(JwtAuthGuard)
async updateProfile(@Req() req, @Body() dto: UpdateProfileDto) {
    // req.user.id est l'ID de l'utilisateur authentifié
    return this.userService.updateProfile(req.user.id, dto);
}
