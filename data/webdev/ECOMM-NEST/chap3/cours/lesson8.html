Étape 1

📝 Inscription et validation d'email

// DTO d'inscription
export class RegisterDto {
  @IsEmail()
  email: string;

  @MinLength(8)
  password: string;

  @IsString()
  firstName: string;

  @IsString()
  lastName: string;
}

// Dans le service
async register(registerDto: RegisterDto) {
  // Vérifier si l'utilisateur existe déjà
  const existingUser = await this.userRepository.findOne({
    where: { email: registerDto.email }
  });
  if (existingUser) {
    throw new ConflictException('Email déjà utilisé');
  }

  // Hachage du mot de passe (voir slide suivante)
  const hashedPassword = await bcrypt.hash(registerDto.password, 10);

  // Création de l'utilisateur
  const user = this.userRepository.create({
    ...registerDto,
    password: hashedPassword,
  });

  await this.userRepository.save(user);
  
  // Envoyer email de confirmation (Nodemailer)
  await this.sendConfirmationEmail(user.email);
  
  return { message: 'Inscription réussie. Vérifiez votre email.' };
}

🔒 Hachage des mots de passe avec bcrypt

import * as bcrypt from 'bcrypt';

// Hachage (salt rounds = 10)
const saltRounds = 10;
const hashedPassword = await bcrypt.hash(password, saltRounds);

// Comparaison lors de la connexion
const isMatch = await bcrypt.compare(plainPassword, hashedPassword);
if (!isMatch) {
  throw new UnauthorizedException('Mot de passe incorrect');
}
✅ Avantages : Hash salé, résistant aux attaques par force brute
⚠️ Important : Ne jamais stocker les mots de passe en clair !
Étape 3

🎫 Génération des tokens JWT

npm install @nestjs/jwt @nestjs/passport passport-jwt

// auth.service.ts
async login(user: User) {
  const payload = { sub: user.id, email: user.email, roles: user.roles };
  
  return {
    access_token: this.jwtService.sign(payload, {
      expiresIn: '15m'  // Access token : 15 minutes
    }),
    refresh_token: this.jwtService.sign(payload, {
      expiresIn: '7d'   // Refresh token : 7 jours
    })
  };
}

// Validation du token
async validateToken(token: string) {
  try {
    return this.jwtService.verify(token);
  } catch (error) {
    throw new UnauthorizedException('Token invalide ou expiré');
  }
}
Compréhension

🔄 Flux d'authentification JWT

1️⃣ Client → envoie email/mot de passe à /auth/login
2️⃣ Serveur → vérifie bcrypt + génère access_token + refresh_token
3️⃣ Client → stocke les tokens (localStorage ou httpOnly cookie)
4️⃣ Client → envoie access_token dans l'en-tête Authorization: Bearer <token>
5️⃣ Serveur → valide le token via JwtStrategy
6️⃣ Refresh → si access_token expire, client utilise refresh_token pour obtenir un nouveau access_token
📘 Bonnes pratiques : Stocker le refresh token en base de données pour pouvoir le révoquer.
Implémentation NestJS

⚙️ JwtStrategy et Guards

// jwt.strategy.ts
@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor(private configService: ConfigService) {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      ignoreExpiration: false,
      secretOrKey: configService.get('JWT_SECRET'),
    });
  }

  async validate(payload: any) {
    return { userId: payload.sub, email: payload.email, roles: payload.roles };
  }
}

// Dans le contrôleur
@UseGuards(JwtAuthGuard)
@Get('profile')
getProfile(@Req() req) {
  return req.user;
}
