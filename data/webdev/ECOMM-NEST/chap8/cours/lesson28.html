📦 Entité Review

Un avis est lié à un produit et à un utilisateur. Il contient une note, un commentaire et éventuellement des images.

// src/reviews/entities/review.entity.ts
import { Entity, Column, PrimaryGeneratedColumn, ManyToOne, CreateDateColumn, UpdateDateColumn } from 'typeorm';
import { Product } from '../../products/entities/product.entity';
import { User } from '../../users/entities/user.entity';

@Entity()
export class Review {
    @PrimaryGeneratedColumn('uuid')
    id: string;

    @Column({ type: 'int', width: 1 })
    rating: number;  // 1 à 5 étoiles

    @Column({ type: 'text' })
    comment: string;

    @Column({ type: 'simple-array', nullable: true })
    images: string[];  // URLs des images uploadées

    @ManyToOne(() => Product, (product) => product.reviews, { onDelete: 'CASCADE' })
    product: Product;

    @ManyToOne(() => User, (user) => user.reviews)
    user: User;

    // Réponse du vendeur/admin
    @Column({ type: 'text', nullable: true })
    sellerResponse: string;

    @Column({ type: 'timestamp', nullable: true })
    respondedAt: Date;

    @CreateDateColumn()
    createdAt: Date;

    @UpdateDateColumn()
    updatedAt: Date;
}
💡 Bonnes pratiques : La note est un entier de 1 à 5. On peut ajouter une validation avec @Min(1) et @Max(5).

✍️ Créer un avis (vérification d'achat)

Un client ne peut laisser un avis que s'il a effectivement acheté le produit.

// src/reviews/reviews.service.ts
async create(userId: string, productId: string, createReviewDto: CreateReviewDto) {
    // 1. Vérifier si l'utilisateur a acheté ce produit
    const hasPurchased = await this.orderService.hasUserPurchasedProduct(userId, productId);
    
    if (!hasPurchased) {
        throw new ForbiddenException('Vous devez acheter le produit avant de laisser un avis');
    }
    
    // 2. Vérifier si un avis existe déjà pour ce produit par cet utilisateur
    const existingReview = await this.reviewRepository.findOne({
        where: { user: { id: userId }, product: { id: productId } }
    });
    
    if (existingReview) {
        throw new ConflictException('Vous avez déjà laissé un avis pour ce produit');
    }
    
    // 3. Créer l'avis
    const review = this.reviewRepository.create({
        ...createReviewDto,
        user: { id: userId },
        product: { id: productId }
    });
    
    return this.reviewRepository.save(review);
}
🔒 Sécurité : La vérification d'achat évite les faux avis et le spam.

✏️ Modifier et supprimer son avis

L'utilisateur ne peut modifier ou supprimer que ses propres avis.

// src/reviews/reviews.service.ts
async update(userId: string, reviewId: string, updateDto: UpdateReviewDto) {
    const review = await this.reviewRepository.findOne({
        where: { id: reviewId },
        relations: ['user']
    });
    
    if (!review) {
        throw new NotFoundException('Avis non trouvé');
    }
    
    if (review.user.id !== userId) {
        throw new ForbiddenException('Vous ne pouvez modifier que vos propres avis');
    }
    
    Object.assign(review, updateDto);
    return this.reviewRepository.save(review);
}

async remove(userId: string, reviewId: string) {
    const review = await this.reviewRepository.findOne({
        where: { id: reviewId },
        relations: ['user']
    });
    
    if (!review) throw new NotFoundException('Avis non trouvé');
    if (review.user.id !== userId) throw new ForbiddenException();
    
    return this.reviewRepository.remove(review);
}
✅ Règle métier : Un utilisateur ne peut pas modifier la note d'un avis après 30 jours (à implémenter).

💬 Répondre aux avis

Les administrateurs et vendeurs peuvent répondre aux avis clients.

// DTO pour la réponse
export class RespondToReviewDto {
    @IsString()
    @IsNotEmpty()
    response: string;
}

// Service
async respondToReview(reviewId: string, sellerResponse: string, userId: string) {
    const review = await this.reviewRepository.findOne({
        where: { id: reviewId },
        relations: ['product']
    });
    
    if (!review) throw new NotFoundException();
    
    // Vérifier que l'utilisateur est admin ou propriétaire du produit (seller)
    const user = await this.userService.findById(userId);
    const isSeller = user.roles.includes('seller') && review.product.sellerId === userId;
    const isAdmin = user.roles.includes('admin');
    
    if (!isAdmin && !isSeller) {
        throw new ForbiddenException('Seuls les admins ou le vendeur peuvent répondre');
    }
    
    review.sellerResponse = sellerResponse;
    review.respondedAt = new Date();
    
    return this.reviewRepository.save(review);
}

🎮 Endpoints du contrôleur

// src/reviews/reviews.controller.ts
@Controller('reviews')
export class ReviewsController {
    @Post()
    @UseGuards(JwtAuthGuard)
    async create(@Req() req, @Body() dto: CreateReviewDto) {
        return this.reviewsService.create(req.user.id, dto.productId, dto);
    }

    @Patch(':id')
    @UseGuards(JwtAuthGuard)
    async update(@Req() req, @Param('id') id: string, @Body() dto: UpdateReviewDto) {
        return this.reviewsService.update(req.user.id, id, dto);
    }

    @Delete(':id')
    @UseGuards(JwtAuthGuard)
    async remove(@Req() req, @Param('id') id: string) {
        return this.reviewsService.remove(req.user.id, id);
    }

    @Post(':id/respond')
    @UseGuards(JwtAuthGuard, RolesGuard)
    @Roles('admin', 'seller')
    async respond(@Param('id') id: string, @Body() dto: RespondToReviewDto, @Req() req) {
        return this.reviewsService.respondToReview(id, dto.response, req.user.id);
    }

    @Get('product/:productId')
    async getProductReviews(@Param('productId') productId: string, @Query() pagination: PaginationDto) {
        return this.reviewsService.findByProduct(productId, pagination);
    }
}

🛒 Vérification d'achat du produit

Méthode du service Order pour vérifier si un utilisateur a acheté un produit.

// src/orders/orders.service.ts
async hasUserPurchasedProduct(userId: string, productId: string): Promise<boolean> {
    const order = await this.orderRepository
        .createQueryBuilder('order')
        .innerJoin('order.items', 'item')
        .where('order.userId = :userId', { userId })
        .andWhere('item.productId = :productId', { productId })
        .andWhere('order.status IN (:...statuses)', { 
            statuses: ['paid', 'shipped', 'delivered'] 
        })
        .getOne();
    
    return !!order;
}
📌 Note : Seules les commandes payées et livrées permettent de laisser un avis.
