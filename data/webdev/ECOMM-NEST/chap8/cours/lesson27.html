📦 Entité Review – structure et décorateurs

// src/reviews/review.entity.ts
import { Entity, Column, PrimaryGeneratedColumn, ManyToOne, CreateDateColumn, UpdateDateColumn } from 'typeorm';
import { User } from '../users/user.entity';
import { Product } from '../products/product.entity';

@Entity('reviews')
export class Review {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ type: 'int', width: 1 })
  rating: number; // 1 à 5 étoiles

  @Column({ type: 'text', nullable: true })
  comment: string;

  @Column({ type: 'simple-array', nullable: true })
  images: string[]; // URLs des images uploadées

  @ManyToOne(() => User, user => user.reviews, { eager: false })
  user: User;

  @ManyToOne(() => Product, product => product.reviews, { onDelete: 'CASCADE' })
  product: Product;

  @CreateDateColumn()
  createdAt: Date;

  @UpdateDateColumn()
  updatedAt: Date;
}

✨ On ajoute aussi dans User et Product la relation inverse @OneToMany.

✅ Vérification : l'utilisateur a-t-il acheté ce produit ?

On interroge la table OrderItem pour savoir si l'utilisateur a une commande avec statut "delivered" contenant ce produit.

// review.service.ts
async userPurchasedProduct(userId: number, productId: number): Promise {
  const orderItem = await this.orderItemRepo
    .createQueryBuilder('item')
    .innerJoin('item.order', 'order')
    .where('order.userId = :userId', { userId })
    .andWhere('item.productId = :productId', { productId })
    .andWhere('order.status = :status', { status: 'delivered' })
    .getOne();
  return !!orderItem;
}

async createReview(userId: number, dto: CreateReviewDto) {
  const hasPurchased = await this.userPurchasedProduct(userId, dto.productId);
  if (!hasPurchased) {
    throw new ForbiddenException('Vous ne pouvez noter que des produits que vous avez achetés et reçus.');
  }
  // ... création de l'avis
}

📌 Bonne pratique :

On peut aussi autoriser l'avis dès que la commande est en statut "paid" (livraison non effectuée), mais la plupart des e-commerces attendent la livraison effective.

📋 DTO & validation des avis

// create-review.dto.ts
import { IsInt, Min, Max, IsString, IsOptional, IsArray, IsUrl } from 'class-validator';
import { ApiProperty } from '@nestjs/swagger';

export class CreateReviewDto {
  @ApiProperty({ minimum: 1, maximum: 5 })
  @IsInt()
  @Min(1)
  @Max(5)
  rating: number;

  @ApiProperty({ required: false })
  @IsString()
  @IsOptional()
  comment?: string;

  @ApiProperty({ required: false, type: [String] })
  @IsArray()
  @IsUrl({}, { each: true })
  @IsOptional()
  images?: string[];

  @ApiProperty()
  @IsInt()
  productId: number;
}

🌐 Endpoints API – Avis clients

// review.controller.ts
@Controller('reviews')
@UseGuards(JwtAuthGuard)
export class ReviewController {
  constructor(private reviewService: ReviewService) {}

  @Post()
  create(@Req() req, @Body() dto: CreateReviewDto) {
    return this.reviewService.createReview(req.user.id, dto);
  }

  @Get('product/:productId')
  getProductReviews(@Param('productId') productId: number) {
    return this.reviewService.getReviewsForProduct(productId);
  }

  @Patch(':id')
  @UseGuards(ReviewOwnerGuard) // Vérifie que l'utilisateur est propriétaire
  update(@Param('id') id: number, @Body() dto: UpdateReviewDto) {
    return this.reviewService.updateReview(id, dto);
  }

  @Delete(':id')
  @UseGuards(ReviewOwnerGuard)
  delete(@Param('id') id: number) {
    return this.reviewService.deleteReview(id);
  }
}

🔐 Le guard ReviewOwnerGuard vérifie que l'utilisateur connecté est bien l'auteur de l'avis (ou admin).

📊 Note moyenne & agrégation

On peut ajouter un champ averageRating dans l'entité Product, mis à jour via un événement ou un cron.

// Dans ReviewService, après création/suppression
async updateProductAverageRating(productId: number) {
  const result = await this.reviewRepo
    .createQueryBuilder('review')
    .select('AVG(review.rating)', 'avg')
    .where('review.productId = :productId', { productId })
    .getRawOne();
  
  const avg = result.avg ? parseFloat(result.avg) : 0;
  await this.productRepo.update(productId, { averageRating: avg, reviewCount: () => `(SELECT COUNT(*) FROM reviews WHERE productId = ${productId})` });
}
💡 Pensez à utiliser un déclencheur SQL ou un événement @AfterInsert/@AfterRemove pour la cohérence.

🧠 Bonnes pratiques pour les avis clients

  • ✅ Ne permettre l'ajout d'avis qu'après un délai raisonnable (ex: 7 jours après livraison).
  • ✅ Utiliser un système de modération automatique pour les commentaires inappropriés (mots interdits).
  • ✅ Permettre aux administrateurs de signaler/supprimer un avis.
  • ✅ Stocker les images sur un CDN avec compression automatique.
  • ✅ Proposer une réponse du vendeur à l'avis (entité ReviewReply).
📈 Les produits avec avis vérifiés et photos convertissent jusqu'à 30% mieux.
