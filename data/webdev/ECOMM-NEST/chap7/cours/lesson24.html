🔌 2. Service PayPal - Connexion SDK

// paypal.service.ts
import * as paypal from '@paypal/checkout-server-sdk';

@Injectable()
export class PaypalService {
    private client: paypal.core.PayPalHttpClient;
    
    constructor(private configService: ConfigService) {
        const environment = this.configService.get('PAYPAL_MODE') === 'live'
            ? new paypal.core.LiveEnvironment(
                this.configService.get('PAYPAL_CLIENT_ID'),
                this.configService.get('PAYPAL_CLIENT_SECRET')
              )
            : new paypal.core.SandboxEnvironment(
                this.configService.get('PAYPAL_CLIENT_ID'),
                this.configService.get('PAYPAL_CLIENT_SECRET')
              );
        this.client = new paypal.core.PayPalHttpClient(environment);
    }
}

🛒 3. Création d'un order PayPal

async createOrder(amount: number, currency: string = 'EUR', returnUrl: string, cancelUrl: string) {
    const request = new paypal.orders.OrdersCreateRequest();
    request.requestBody({
        intent: 'CAPTURE',
        purchase_units: [{
            amount: { currency_code: currency, value: amount.toFixed(2) }
        }],
        application_context: {
            return_url: returnUrl,
            cancel_url: cancelUrl,
            brand_name: "Ma Boutique Cameroun",
            landing_page: "BILLING",
            user_action: "PAY_NOW"
        }
    });
    
    const response = await this.client.execute(request);
    const approvalLink = response.result.links.find(link => link.rel === 'approve').href;
    return { orderId: response.result.id, approvalUrl: approvalLink };
}

🌐 4. Endpoint de création

// payment.controller.ts
@Post('paypal/create-order')
@UseGuards(JwtAuthGuard)
async createPayPalOrder(@Req() req, @Body() body: { amount: number }) {
    const returnUrl = `${this.configService.get('FRONTEND_URL')}/payment/success`;
    const cancelUrl = `${this.configService.get('FRONTEND_URL')}/payment/cancel`;
    
    const { orderId, approvalUrl } = await this.paypalService.createOrder(
        body.amount, 'EUR', returnUrl, cancelUrl
    );
    
    // Stocker orderId en session ou en base avec l'utilisateur
    await this.paymentService.savePendingPayment(req.user.id, orderId, body.amount);
    
    return { approvalUrl, orderId };
}

✅ 5. Capture de l'order

async captureOrder(orderId: string) {
    const request = new paypal.orders.OrdersCaptureRequest(orderId);
    request.requestBody({});
    
    const response = await this.client.execute(request);
    const capture = response.result;
    
    if (capture.status === 'COMPLETED') {
        return {
            success: true,
            transactionId: capture.purchase_units[0].payments.captures[0].id,
            amount: capture.purchase_units[0].payments.captures[0].amount.value,
            status: capture.status
        };
    }
    throw new BadRequestException('Paiement non complété');
}

🔁 6. Route de retour (success)

@Get('paypal/success')
async paypalSuccess(@Query('token') orderId: string, @Query('PayerID') payerId: string) {
    // Le token = orderId PayPal
    const captureResult = await this.paypalService.captureOrder(orderId);
    
    if (captureResult.success) {
        // Mettre à jour la commande locale
        await this.orderService.markAsPaid(orderId, captureResult.transactionId);
        
        // Vider le panier
        await this.cartService.clearCart(userId);
        
        return { message: 'Paiement réussi', transactionId: captureResult.transactionId };
    }
    throw new BadRequestException('Erreur lors de la capture');
}

❌ 7. Gestion de l'annulation

@Get('paypal/cancel')
async paypalCancel(@Query('token') orderId: string) {
    // Annulation côté client : on peut supprimer la session ou simplement rediriger
    await this.paymentService.markAsCancelled(orderId);
    
    return { message: 'Paiement annulé par l\'utilisateur' };
}

// Service payment
async markAsCancelled(orderId: string) {
    // On ne touche pas au stock, la commande n'a pas été validée
    await this.paymentRepository.update(
        { paypalOrderId: orderId },
        { status: 'CANCELLED_BY_USER' }
    );
}
