⭐ Store des favoris (local)

On crée un store Zustand avec persistance pour gérer la liste des produits favoris de l'utilisateur, même hors ligne.

// stores/favoritesStore.js
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';

const useFavoritesStore = create(
  persist(
    (set, get) => ({
      favorites: [], // Liste des IDs ou objets produits
      toggleFavorite: (product) => {
        const exists = get().favorites.some(p => p.id === product.id);
        let newFavorites;
        if (exists) {
          newFavorites = get().favorites.filter(p => p.id !== product.id);
        } else {
          newFavorites = [...get().favorites, product];
        }
        set({ favorites: newFavorites });
      },
      isFavorite: (productId) => get().favorites.some(p => p.id === productId),
      syncFromBackend: (backendFavorites) => set({ favorites: backendFavorites }),
      clearFavorites: () => set({ favorites: [] }),
    }),
    {
      name: 'favorites-storage',
      storage: createJSONStorage(() => AsyncStorage),
    }
  )
);

export default useFavoritesStore;
📌 Note : On stocke les objets complets ou seulement les IDs selon les besoins. Pour une synchro efficace, on peut ne stocker que les IDs.

🔄 Synchronisation backend

Lorsque l'utilisateur se connecte, on doit fusionner les favoris locaux avec ceux du serveur (généralement on prend l'intersection ou on privilégie le serveur).

// Synchronisation après login
import useFavoritesStore from './stores/favoritesStore';
import api from './services/api';

async function syncFavoritesWithBackend() {
  const { favorites, syncFromBackend } = useFavoritesStore.getState();
  
  try {
    // 1. Envoyer les favoris locaux vers le backend (merge)
    await api.post('/favorites/sync', { favorites });
    
    // 2. Récupérer la version serveur à jour
    const response = await api.get('/favorites');
    syncFromBackend(response.data);
    
    console.log('✅ Favoris synchronisés');
  } catch (error) {
    console.error('Erreur synchro favoris', error);
  }
}

📲 Notifications push avec Expo

Expo Notifications permet d'envoyer des notifications même lorsque l'app est en arrière-plan. Pour les promotions sur les favoris, le backend peut cibler les utilisateurs ayant certains produits en favoris.

// Installation
expo install expo-notifications expo-device

// Demander les permissions et récupérer le token push
import * as Notifications from 'expo-notifications';
import * as Device from 'expo-device';

async function registerForPushNotificationsAsync() {
  if (!Device.isDevice) return;
  
  const { status: existingStatus } = await Notifications.getPermissionsAsync();
  let finalStatus = existingStatus;
  
  if (existingStatus !== 'granted') {
    const { status } = await Notifications.requestPermissionsAsync();
    finalStatus = status;
  }
  
  if (finalStatus !== 'granted') return;
  
  const token = (await Notifications.getExpoPushTokenAsync()).data;
  console.log('📱 Expo Push Token:', token);
  
  // Envoyer ce token à votre backend associé à l'utilisateur
  await api.post('/users/push-token', { token });
  
  return token;
}

🔔 Réception et gestion des notifications

Mettre en place un écouteur pour réagir aux notifications (ex: ouvrir l'écran du produit en promotion).

// App.js ou composant racine
import { useEffect } from 'react';
import * as Notifications from 'expo-notifications';

Notifications.setNotificationHandler({
  handleNotification: async () => ({
    shouldShowAlert: true,
    shouldPlaySound: true,
    shouldSetBadge: false,
  }),
});

export default function App() {
  useEffect(() => {
    const subscription = Notifications.addNotificationReceivedListener(notification => {
      console.log('Notification reçue en avant-plan', notification);
    });

    const responseSubscription = Notifications.addNotificationResponseReceivedListener(response => {
      const productId = response.notification.request.content.data.productId;
      // Navigation vers l'écran du produit
      navigation.navigate('Product', { id: productId });
    });

    return () => {
      subscription.remove();
      responseSubscription.remove();
    };
  }, []);

  // ... reste de l'app
}
🚀 Astuce : Utilisez le champ `data` de la notification pour transmettre des informations (productId, screen, etc.).

🏷️ Promotions sur les favoris

Une approche courante est d'ajouter un champ `onSale` ou `discount` aux produits, puis de vérifier si un produit favori est en promotion pour déclencher une notification.

📱 Côté mobile
  • Store des favoris (local/serveur)
  • Polling ou WebSocket pour détecter promo
  • Affichage d'un badge 🔥 sur le produit
🖥️ Côté serveur
  • Planification de tâches (cron jobs)
  • Vérification des nouveaux produits en promo
  • Envoi groupé aux tokens des utilisateurs concernés

Le taux de conversion via notifications push sur favoris est généralement 2 à 3 fois plus élevé qu'une publicité classique.
