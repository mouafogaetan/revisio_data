1. FlatList : la liste performante

FlatList est le composant natif pour les grandes listes. Il ne rend que les éléments visibles à l'écran.

import { FlatList, View, Text, Image } from 'react-native';

function ProductList({ products }) {
  const renderItem = ({ item }) => (
    <View className="p-4 border-b border-gray-200">
      <Image source={{ uri: item.image }} className="w-20 h-20" />
      <Text className="font-bold">{item.name}</Text>
      <Text>{item.price} FCFA</Text>
    </View>
  );

  return (
    <FlatList
      data={products}
      renderItem={renderItem}
      keyExtractor={(item) => item.id.toString()}
      showsVerticalScrollIndicator={false}
      initialNumToRender={10}
      maxToRenderPerBatch={10}
      windowSize={21}
    />
  );
}
⚡ Optimisations clés : initialNumToRender, maxToRenderPerBatch, windowSize pour éviter les saccades.

2. Scroll infini (pagination)

import { useState } from 'react';
import { FlatList, ActivityIndicator } from 'react-native';

function InfiniteProductList() {
  const [products, setProducts] = useState([]);
  const [page, setPage] = useState(1);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);

  const loadMoreProducts = async () => {
    if (loading || !hasMore) return;
    setLoading(true);
    const response = await fetch(`/api/products?page=${page}&limit=20`);
    const newProducts = await response.json();
    setProducts(prev => [...prev, ...newProducts]);
    setPage(prev => prev + 1);
    setHasMore(newProducts.length === 20);
    setLoading(false);
  };

  return (
    <FlatList
      data={products}
      renderItem={({ item }) => <ProductCard product={item} />}
      keyExtractor={item => item.id}
      onEndReached={loadMoreProducts}
      onEndReachedThreshold={0.5}
      ListFooterComponent={loading && <ActivityIndicator size="large" />}
    />
  );
}
📌 onEndReachedThreshold : déclenche le chargement quand l'utilisateur est à 50% de la fin de la liste.

3. Filtres avec BottomSheet

On utilise @gorhom/bottom-sheet pour un modal glissant depuis le bas.

npm install @gorhom/bottom-sheet react-native-reanimated react-native-gesture-handler
import BottomSheet from '@gorhom/bottom-sheet';
import { useRef, useState } from 'react';

function FilterSheet({ visible, onClose, onApply }) {
  const sheetRef = useRef(null);
  const [priceRange, setPriceRange] = useState([0, 100000]);
  const [selectedCategory, setSelectedCategory] = useState('');

  return (
    <BottomSheet
      ref={sheetRef}
      index={visible ? 0 : -1}
      snapPoints={['50%']}
      onClose={onClose}
    >
      <View className="p-4">
        <Text className="text-lg font-bold">Filtrer</Text>
        <Text>Prix max : {priceRange[1]} FCFA</Text>
        <Slider value={priceRange[1]} onValueChange={v => setPriceRange([0, v])} />
        
        <Text>Catégorie</Text>
        <Picker selectedValue={selectedCategory} onValueChange={setSelectedCategory}>
          <Picker.Item label="Toutes" value="" />
          <Picker.Item label="Électronique" value="electronics" />
          <Picker.Item label="Mode" value="fashion" />
        </Picker>
        
        <Button title="Appliquer" onPress={() => onApply({ priceRange, selectedCategory })} />
      </View>
    </BottomSheet>
  );
}
✨ snapPoints définit la hauteur du BottomSheet (ex: 50% de l'écran).

4. Recherche avec debounce

import { useState, useEffect } from 'react';
import { Searchbar } from 'react-native-paper';

function SearchProducts() {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    const delayDebounce = setTimeout(() => {
      if (query.length >= 2) {
        fetch(`/api/search?q=${query}`)
          .then(res => res.json())
          .then(setResults);
      } else {
        setResults([]);
      }
    }, 300);

    return () => clearTimeout(delayDebounce);
  }, [query]);

  return (
    <View>
      <Searchbar
        placeholder="Rechercher un produit..."
        value={query}
        onChangeText={setQuery}
      />
      <FlatList data={results} renderItem={({ item }) => <ProductCard product={item} />} />
    </View>
  );
}
⏱️ Le debounce (300ms) évite d'envoyer une requête à chaque frappe, économisant les ressources serveur.

5. Combinaison : état global des filtres

On utilise un store Zustand pour centraliser les filtres (catégorie, prix, tri).

// store/filterStore.js
import { create } from 'zustand';

export const useFilterStore = create((set) => ({
  category: '',
  priceMax: 100000,
  sortBy: 'popularity',
  searchQuery: '',
  setCategory: (category) => set({ category }),
  setPriceMax: (priceMax) => set({ priceMax }),
  setSortBy: (sortBy) => set({ sortBy }),
  setSearchQuery: (searchQuery) => set({ searchQuery }),
  resetFilters: () => set({ category: '', priceMax: 100000, sortBy: 'popularity', searchQuery: '' }),
}));
// Dans l'écran produit
const filters = useFilterStore();
const { data: products } = useQuery(['products', filters], () => fetchProducts(filters));
✅ Tous les filtres sont réactifs : dès qu'un filtre change, la requête se relance automatiquement.
